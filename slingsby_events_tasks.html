    } 

    cal_insert_response = service.calendars().insert(body=calendar).execute() 

    public_acl = { 

        'role': 'reader', 

        'scope': { 

            'type': 'default' 

        } 

    } 

    acl_insert_response = service.acl().insert(calendarId=cal_insert_response['id'], body=public_acl).execute() 

    return acl_insert_response 

 

 

def get_calendar_service(): 

    name = 'calendar' 

    version = 'v3' 

    scope = 'https://www.googleapis.com/auth/calendar' 

 

    # Prepare credentials, and authorize HTTP object with them. 

    with open('api-credentials.json') as fh: 

        data = json.load(fh) 

    credentials = SignedJwtAssertionCredentials(data['client_email'], data['private_key'], scope) 

    http = credentials.authorize(http=httplib2.Http()) 

 

    # Construct a service object via the discovery service. 

    service = discovery.build(name, version, http=http) 

    return service
